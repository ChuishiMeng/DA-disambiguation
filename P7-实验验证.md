### 2.1 歧义检测器

```python
# core/detector.py

from typing import List, Tuple, Dict
from dataclasses import dataclass
from enum import Enum

class AmbiguityType(Enum):
    USER_EXPRESSION = "user_expression"      # 用户层：表达习惯
    USER_ROLE = "user_role"                   # 用户层：角色视角
    BUSINESS_CONCEPT = "business_concept"     # 业务层：概念定义
    BUSINESS_RULE = "business_rule"           # 业务层：计算规则
    BUSINESS_TIME = "business_time"           # 业务层：时间约定
    DATA_TABLE = "data_table"                 # 数据层：表映射
    DATA_COLUMN = "data_column"               # 数据层：列映射
    DATA_VALUE = "data_value"                 # 数据层：值解析

@dataclass
class Ambiguity:
    type: AmbiguityType
    layer: str  # "user" | "business" | "data"
    concept: str
    description: str
    candidates: List[Dict]
    prior_prob: float
    confidence: float

class AmbiguityDetector:
    """歧义检测器"""
    
    def __init__(self, llm_client, rule_engine):
        self.llm = llm_client
        self.rules = rule_engine
    
    def detect(self, query: str, context: Dict) -> List[Ambiguity]:
        """
        检测查询中的潜在歧义
        
        返回:
            List[Ambiguity]: 歧义列表
        """
        ambiguities = []
        
        # 方法 1: LLM 分类
        llm_ambiguities = self._llm_detect(query, context)
        ambiguities.extend(llm_ambiguities)
        
        # 方法 2: 规则匹配
        rule_ambiguities = self._rule_detect(query, context)
        ambiguities.extend(rule_ambiguities)
        
        # 方法 3: Schema 差异检测
        schema_ambiguities = self._schema_detect(query, context)
        ambiguities.extend(schema_ambiguities)
        
        # 去重 + 合并
        return self._merge_ambiguities(ambiguities)
    
    def _llm_detect(self, query: str, context: Dict) -> List[Ambiguity]:
        """LLM 分类检测"""
        prompt = f"""
        分析以下查询中的潜在歧义：
        
        查询：{query}
        上下文：{context}
        
        请识别：
        1. 可能有多重解释的概念
        2. 每个概念的候选解释
        3. 各解释的先验概率
        
        输出 JSON 格式。
        """
        response = self.llm.generate(prompt)
        return self._parse_llm_response(response)
    
    def _rule_detect(self, query: str, context: Dict) -> List[Ambiguity]:
        """规则匹配检测"""
        ambiguities = []
        
        # 检查歧义关键词
        for concept, rules in self.rules.ambiguity_patterns.items():
            if concept in query:
                for rule in rules:
                    if rule.match(query):
                        amb = Ambiguity(
                            type=rule.ambiguity_type,
                            layer=rule.layer,
                            concept=concept,
                            description=rule.description,
                            candidates=rule.candidates,
                            prior_prob=rule.prior,
                            confidence=0.7
                        )
                        ambiguities.append(amb)
        
        return ambiguities
    
    def _schema_detect(self, query: str, context: Dict) -> List[Ambiguity]:
        """Schema 差异检测"""
        # 生成多个候选 SQL
        candidate_sqls = self._generate_candidate_sqls(query, context)
        
        if len(candidate_sqls) < 2:
            return []
        
        # 分析 SQL 差异
        differences = self._analyze_sql_differences(candidate_sqls)
        
        # 转换为歧义
        ambiguities = []
        for diff in differences:
            amb = Ambiguity(
                type=diff.type,
                layer="data",
                concept=diff.concept,
                description=diff.description,
                candidates=diff.candidates,
                prior_prob=0.5,
                confidence=diff.confidence
            )
            ambiguities.append(amb)
        
        return ambiguities
    
    def _merge_ambiguities(self, ambiguities: List[Ambiguity]) -> List[Ambiguity]:
        """合并重复歧义"""
        merged = {}
        for amb in ambiguities:
            key = (amb.layer, amb.concept)
            if key not in merged:
                merged[key] = amb
            else:
                # 置信度平均
                merged[key].confidence = (merged[key].confidence + amb.confidence) / 2
        
        return list(merged.values())
```

# core/user_layer.py

class UserLayerInference:
    """用户层推理"""
    
    def __init__(self, profile_manager):
        self.profile_manager = profile_manager
    
    def infer(self, query: str, user_id: str, context: Dict) -> Tuple[Dict, float]:
        """
        用户层推理
        
        返回:
            (evidence, confidence): 证据分布和置信度
        """
        # 获取用户画像
        profile = self.profile_manager.get_profile(user_id)
        
        if profile is None:
            # 冷启动
            return self._cold_start_inference(query, context)
        
        # 提取查询特征
        features = self._extract_query_features(query)
        
        # 匹配用户偏好
        preferences = self._match_preferences(features, profile)
        
        # 计算置信度
        confidence = self._compute_confidence(preferences, profile)
        
        return preferences, confidence
    
    def _cold_start_inference(self, query: str, context: Dict) -> Tuple[Dict, float]:
        """冷启动推理"""
        # 基于查询内容推断角色
        role = self._infer_role_from_query(query)
        
        # 使用角色先验
        prior = self._get_role_prior(role)
        
        return prior, 0.5  # 低置信度


# core/business_layer.py

class BusinessLayerInference:
    """业务层推理"""
    
    def __init__(self, knowledge_graph):
        self.kg = knowledge_graph
    
    def match(self, query: str, context: Dict) -> Tuple[Dict, float]:
        """
        业务层知识匹配
        
        返回:
            (matched_rules, confidence): 匹配的规则和置信度
        """
        # 实体抽取
        entities = self._extract_entities(query)
        
        matched_rules = {}
        confidences = []
        
        for entity in entities:
            # 图谱检索
            concept_node = self.kg.find_concept(entity)
            
            if concept_node:
                # 获取关联规则
                rules = concept_node.get_computed_by_rules()
                
                # 规则过滤
                filtered = self._filter_rules(rules, context)
                
                # 计算置信度
                for rule in filtered:
                    conf = self._compute_rule_confidence(rule, context)
                    confidences.append(conf)
                    matched_rules[entity] = rule
        
        avg_confidence = np.mean(confidences) if confidences else 0.0
        
        return matched_rules, avg_confidence


# core/data_layer.py

class DataLayerInference:
    """数据层推理"""
    
    def __init__(self, schema_metadata, llm_client):
        self.schema = schema_metadata
        self.llm = llm_client
    
    def link(self, query: str, context: Dict) -> Tuple[Dict, float]:
        """
        Schema Linking
        
        返回:
            (schema_mapping, confidence): Schema 映射和置信度
        """
        # 实体识别
        entities = self._extract_entities(query)
        
        mappings = {}
        confidences = []
        
        for entity in entities:
            # 候选生成
            candidates = self._generate_candidates(entity, self.schema)
            
            if not candidates:
                continue
            
            # 候选排序
            ranked = self._rank_candidates(entity, candidates)
            
            # 选择最佳
            best = ranked[0]
            mappings[entity] = best
            
            # 计算置信度
            conf = self._compute_linking_confidence(best, ranked)
            confidences.append(conf)
        
        avg_confidence = np.mean(confidences) if confidences else 0.0
        
        return mappings, avg_confidence

### 2.3 协同决策引擎

```python
# core/collaborative.py

class CollaborativeDecisionEngine:
    """协同决策引擎"""
    
    def __init__(self, config):
        self.config = config
        self.thresholds = config['thresholds']
    
    def decide(self, user_evidence, business_evidence, data_evidence,
               user_conf, business_conf, data_conf,
               context: Dict) -> Dict:
        """
        协同决策
        
        返回:
            decision: 决策结果
        """
        # Step 1: 置信度校准
        user_conf_cal = self._calibrate(user_conf, temp=1.5)
        business_conf_cal = self._calibrate(business_conf, temp=1.2)
        data_conf_cal = self._calibrate(data_conf, temp=1.0)
        
        # Step 2: 动态权重计算
        weights = self._compute_weights(
            context['query_type'],
            user_conf_cal, business_conf_cal, data_conf_cal,
            context
        )
        
        # Step 3: 证据融合
        joint_posterior = self._fuse_evidence(
            user_evidence, business_evidence, data_evidence,
            weights
        )
        
        # Step 4: 歧义阻断检查
        blocked, blocking_layer, block_conf = self._check_blocking(
            user_conf_cal, business_conf_cal, data_conf_cal
        )
        
        if blocked:
            # 某层高置信度，直接决策
            best_intent = self._get_best_from_layer(
                blocking_layer,
                user_evidence if blocking_layer == 'user' else
                business_evidence if blocking_layer == 'business' else
                data_evidence
            )
            return {
                'action': 'execute',
                'intent': best_intent,
                'confidence': block_conf,
                'explanation': f'{blocking_layer}层高置信度阻断'
            }
        
        # Step 5: 决策
        best_intent = max(joint_posterior, key=joint_posterior.get)
        max_conf = joint_posterior[best_intent]
        
        if max_conf > self.thresholds['execute']:
            return {
                'action': 'execute',
                'intent': best_intent,
                'confidence': max_conf,
                'explanation': self._build_explanation(best_intent, joint_posterior)
            }
        else:
            # 需要交互确认
            return {
                'action': 'clarify',
                'intent': best_intent,
                'confidence': max_conf,
                'candidates': self._get_top_k(joint_posterior, k=3),
                'explanation': '置信度不足，需要确认'
            }
    
    def _compute_weights(self, query_type, user_conf, business_conf, data_conf, context):
        """动态权重计算"""
        # 基础权重
        base = {'user': 0.33, 'business': 0.34, 'data': 0.33}
        
        # 查询类型调整
        if query_type == '个人业绩':
            base['user'] += 0.2
        elif query_type == '业务指标':
            base['business'] += 0.2
        elif query_type == '技术探索':
            base['data'] += 0.2
        
        # 置信度调整 (Softmax)
        confs = np.array([user_conf, business_conf, data_conf])
        conf_weights = softmax(confs)
        
        # 融合
        alpha = 0.6
        final = {}
        keys = ['user', 'business', 'data']
        for i, key in enumerate(keys):
            final[key] = alpha * base[key] + (1 - alpha) * conf_weights[i]
        
        # 归一化
        total = sum(final.values())
        return tuple(final[k] / total for k in keys)
    
    def _fuse_evidence(self, user_ev, business_ev, data_ev, weights):
        """证据融合：加权几何平均"""
        w_u, w_b, w_d = weights
        
        all_intents = set(user_ev.keys()) | set(business_ev.keys()) | set(data_ev.keys())
        uniform = 1.0 / len(all_intents)
        
        joint = {}
        for intent in all_intents:
            p_u = user_ev.get(intent, uniform)
            p_b = business_ev.get(intent, uniform)
            p_d = data_ev.get(intent, uniform)
            
            joint[intent] = (p_u ** w_u) * (p_b ** w_b) * (p_d ** w_d)
        
        # 归一化
        total = sum(joint.values())
        return {k: v / total for k, v in joint.items()}
```

# core/interaction.py

class InteractionStrategy:
    """交互策略"""
    
    def __init__(self, llm_client, question_templates):
        self.llm = llm_client
        self.templates = question_templates
        self.patience_model = UserPatienceModel()
    
    def select_question(self, posterior: Dict, ambiguities: List,
                       user_id: str, context: Dict) -> str:
        """
        选择最优澄清问题
        
        返回:
            question: 自然语言问题
        """
        # 获取用户耐心
        patience = self.patience_model.get_patience(user_id, context)
        
        # 生成候选问题
        candidates = []
        for amb in ambiguities:
            for template in self.templates[amb.type]:
                q = self._instantiate_template(template, amb)
                eig = self._compute_eig(q, posterior)
                candidates.append((q, eig, amb))
        
        # 排序
        candidates.sort(key=lambda x: x[1], reverse=True)
        
        # 考虑耐心
        if patience < 0.3:
            # 耐心不足：单轮最优
            best = candidates[0]
        else:
            # 耐心充足：多轮策略
            best = self._select_multi_round_strategy(candidates, posterior)
        
        # 自然度优化
        natural_question = self._naturalize(best[0])
        
        return natural_question
    
    def _compute_eig(self, question, posterior):
        """计算期望信息增益"""
        # H[P(意图│证据)]
        prior_entropy = self._entropy(posterior)
        
        # Eₐ[H[P(意图│证据，回答=a)]]
        expected_posterior_entropy = 0.0
        for answer in self._get_possible_answers(question):
            # 模拟回答后的后验
            updated = self._update_posterior(posterior, question, answer)
            prob_answer = self._prob_answer(question, answer, posterior)
            expected_posterior_entropy += prob_answer * self._entropy(updated)
        
        # EIG
        eig = prior_entropy - expected_posterior_entropy
        
        return eig
```

# experiments/main_experiment.py

import json
from evaluators import DAEvaluator
from models import TriCoDAG, Odin, LinkAlign, AmbiSQL

def run_main_experiment():
    """主实验"""
    
    # 加载数据
    test_data = load_dataset('test')
    
    # 初始化模型
    models = {
        'TriCo-DAG': TriCoDAG(),
        'Odin': Odin(),
        'LinkAlign': LinkAlign(),
        'AmbiSQL': AmbiSQL(),
    }
    
    # 初始化评估器
    evaluator = DAEvaluator()
    
    results = {}
    
    for model_name, model in models.items():
        print(f"Running {model_name}...")
        
        predictions = []
        for query in test_data:
            pred = model.predict(query)
            predictions.append(pred)
        
        # 评估
        metrics = evaluator.evaluate(predictions, test_data)
        results[model_name] = metrics
        
        # 保存中间结果
        save_results(f'results/{model_name}.json', metrics)
    
    # 统计检验
    comparisons = {}
    for model_name in models:
        if model_name != 'TriCo-DAG':
            comp = evaluator.statistical_test(
                results['TriCo-DAG'],
                results[model_name]
            )
            comparisons[model_name] = comp
    
    # 保存最终结果
    save_results('results/main_experiment.json', {
        'metrics': results,
        'comparisons': comparisons
    })
    
    # 生成报告
    generate_report(results, comparisons)
    
    return results
```

# scripts/visualize.py

import matplotlib.pyplot as plt
import seaborn as sns

def plot_main_results(results):
    """主实验结果可视化"""
    
    methods = list(results.keys())
    da_scores = [results[m]['da_score'] for m in methods]
    
    plt.figure(figsize=(10, 6))
    bars = plt.bar(methods, da_scores)
    
    # 标注显著性
    for i, method in enumerate(methods):
        if method != 'TriCo-DAG':
            p_value = results[method]['p_value']
            if p_value < 0.001:
                plt.text(i, da_scores[i] + 0.02, '***', ha='center')
            elif p_value < 0.01:
                plt.text(i, da_scores[i] + 0.02, '**', ha='center')
            elif p_value < 0.05:
                plt.text(i, da_scores[i] + 0.02, '*', ha='center')
    
    plt.ylabel('DA-Score')
    plt.title('Main Experiment Results')
    plt.xticks(rotation=45)
    plt.tight_layout()
    plt.savefig('figures/main_results.png', dpi=300)
